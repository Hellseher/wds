#!/usr/bin/env bash
# File : fedora-uprade-release.sh

if [ $USER != "root" ]
then
    echo "Run as root or sudo user. Exit"
fi

REQUIRE=(
    grep
)

chk_fedora()
{ # Check if we are on Fedora distro
    echo tf a
}


main()
{
    dnf upgrade --refresh
    dnf install dnf-plugin-system-upgrade
    dnf system-upgrade download --releasever=33
    dnf system-upgrade reboot
}
main "$@"
